uranium_expenditure_buffer = { # Buffer, will not be spent
	resource = uranium
	type = expenditure
	category = buffer

	potential = {
		always = no
		btr_ai_budget_potential_uranium_expenditure_buffer = yes
	}

	weight = {
		weight = 0.0
	}
}

uranium_expenditure_ships = {
	resource = uranium
	type = expenditure
	category = ships

	potential = {
		btr_ai_budget_potential_uranium_expenditure_ships = yes
	}

	weight = {
		weight = 0.6

		# spend more alloys building ships while at war
		modifier = {
			factor = 2
			OR = {
				recently_lost_war = yes
				is_at_war = yes
			}
		}

		# If we want to colonize and are at peace then prio that
		modifier = {
			factor = 0.25
			AND = {
				ai_colonize_plans > 0
				is_at_war = no
			}
		}

		# spend less alloys on ships if over capacity
		modifier = {
			factor = 0.5
			used_naval_capacity_percent >= 1.0
		}

		modifier = {
			factor = 5
			used_naval_capacity_percent < 1.0
			OR = {
				any_neighbor_country = {
					has_ascension_perk = ap_become_the_crisis
				}
				any_country = {
					is_crisis_faction = yes
				}
			}
		}
		modifier = {
			factor = 10
			years_passed < 50
			count_owned_fleet = {
				count < value:desired_science_ships
				limit = {
					is_ship_class = shipclass_science_ship
				}
			}
		}
	}

	desired_min = {
		base = 150
	}
}

uranium_expenditure_ship_upgrades = {
	resource = uranium
	type = expenditure
	category = ship_upgrades

	potential = {
		btr_ai_budget_potential_uranium_expenditure_ship_upgrades = yes
		is_at_war = no
		any_owned_fleet = {
			controller = { is_same_value = root }
			can_be_upgraded = yes
		}
	}

	weight = {
		weight = 0.2

		# spend more on ship upgrades if at capacity
		modifier = {
			factor = 2.5
			used_naval_capacity_percent >= 1.0
		}
	}
}

	type = upkeep
	category = country

	potential = {
		always = yes
	}

	weight = {
		weight = 1

			factor = 1.25

			}
		}
		modifier = {
			factor = 1.5
