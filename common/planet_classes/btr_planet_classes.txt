pc_swamp = {
	entity = "swamp_planet"

	icon = GFX_planet_type_swamp

	picture = pc_swamp
	atmosphere_color 		= hsv { 0.58 0.3 0.7 }

	spawn_odds = 0
	uses_alternative_skies_for_moons = no
	district_set = standard
	carry_cap_per_free_district = @carry_cap_normal
}
