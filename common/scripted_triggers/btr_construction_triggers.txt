

btr_enable_clerks = {
	owner = {
		is_regular_empire = yes
		or = {
			is_ai = no
			prev = { has_planet_flag = btr_should_use_clerks }
			years_passed < 1
			and = {
				btr_raw_ecocrisis_level_1 = yes
				prev = {
					solar_system = {
						exists = starbase
						starbase = {
							has_starbase_size > starbase_outpost
						}
					}
					has_planet_flag = btr_need_clerks
				}
			}
			and = {
				years_passed < 9
				years_passed > 3
			}
			and = {
				years_passed > 60
				not = { has_country_flag = btr_has_room_to_grow }
				prev = {
					free_building_slots = 0
					free_district_slots = 0
				}
			}
		}
	}
}

# Defines if ai_budget should potential for habitats, adjustable via DMM (event btr_menu.1). Mods can hook up to these options by checking for flags in their megastructure potential or possible block.
btr_ai_should_build_habitats = {
	OR = {
		has_country_flag = btr_country_ai_should_build_habitats		# Exception for global flag which bans habitats.
		NOR = {
			has_global_flag = btr_global_ai_should_not_build_habitats
			AND = {
				has_country_flag = btr_country_ai_should_not_build_habitats
				btr_is_habitat_empire = no 		# Habitat empires should build habitats anyway.
			}
		}
	}
}

btr_is_habitat_empire = {
	OR = {
		has_origin = origin_void_dwellers
		has_country_flag = btr_country_is_habitat_empire
	}
