can_have_light_metal_deposits = {
	OR = {
		is_planet_class = pc_carbon
		is_planet_class = pc_iron
		is_planet_class = pc_chthonian
		is_planet_class = pc_hothouse
		#is_planet_class = pc_vulcanoid # No resources at all
		is_planet_class = pc_met_asteroid # Metals Only
		#is_planet_class = pc_sil_asteroid # Minerals Only
		is_planet_class = pc_carb_asteroid # Light Metals, Minerals, and rare Eezo 
		#is_planet_class = pc_asteroid 
		#is_planet_class = pc_gel_asteroid # Hydrogen Only
		#is_planet_class = pc_eezo_asteroid # Eezo Only
		is_planet_class = pc_molten
		is_planet_class = pc_barren
		is_planet_class = pc_barren_cold
		#is_planet_class = pc_toxic # Disabled in BTR
		is_planet_class = pc_frozen
		is_planet_class = pc_broken
		is_planet_class = pc_shattered
	}
}
