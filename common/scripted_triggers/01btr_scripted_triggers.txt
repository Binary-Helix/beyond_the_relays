# Evaluates to true if btrai fork is actually running. Also see GetName_btr_or_tech_fork_switch scripted localisation.
btr_is_btrai_loaded = {
	event_target:global_event_country = {
		check_variable_arithmetic = {
			which = @btr_or_tech_fork_switch			# The following variable can be used to determine whether btr (1) or btrai (2) is loaded. Always trigger check with equal.
			value = 2
		}
	}
}

# The following variable can be used to determine if btr Friendship Patch is properly loaded (overwriting btr). Always trigger check with equal. It does not check if other mods are overwriting the patch.
btr_is_friendship_patch_loaded = {
	event_target:global_event_country = {
		check_variable_arithmetic = {
			which = @btr_is_friendship_patch_loaded
			value = 1
		}
	}
}
