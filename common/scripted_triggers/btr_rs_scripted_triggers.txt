is_hot_zone = {
	OR = { 
		has_planet_flag = hot_zone
		AND = {
			has_planet_flag = warm_zone
			solar_system = {
				OR = { 
					is_red_giant = yes
					is_yellow_giant = yes
					is_red_supergiant = yes
					is_yellow_supergiant = yes
					is_blue_supergiant = yes
				}
			}
		}
		AND = {
			has_planet_flag = temperate_zone
			solar_system = {
				OR = { 
					is_red_supergiant = yes
					is_yellow_supergiant = yes
					is_blue_supergiant = yes
				}
			}
		}
	}
}
