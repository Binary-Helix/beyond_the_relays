building_military_academy = {
	base_buildtime = @btr_mili_b2_time
	base_cap_amount = 1
	can_be_ruined = no
	category = army

	potential = {
		exists = owner
		owner = { NOT = { has_valid_civic = civic_reanimated_armies } }
		NOT = { has_planet_flag = planetary_station }
	}

	allow = {
		has_upgraded_capital = yes
		if = {
			limit = {
				exists = owner
				owner = { is_ai = yes }
			}
			OR = {
				has_building = building_stronghold
				has_building = building_fortress
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = {
			OR = {
				has_valid_civic = civic_reanimated_armies
				AND = {
					is_ai = yes
					prev = {
						NOR = {
							has_building = building_stronghold
							has_building = building_fortress
						}
					}
				}
			}
		}
	}

	convert_to = {
		building_dread_encampment
	}

	army_modifier = {
		army_starting_experience_add = 100
	}

	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = no }
		}
		modifier = {
			job_soldier_add = @b1_jobs
		}
	}
	triggered_planet_modifier = {
		potential = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		modifier = {
			job_warrior_drone_add = @b1_jobs
		}
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @btr_mili_b1_mineral_cost
			light_metals = @btr_mili_b1_light_metals_cost
			heavy_metals = @btr_mili_b1_heavy_metals_cost
		}
		upkeep = {
			energy = @btr_mili_b1_energy_upkeep
			uranium = @btr_mili_b1_uranium_upkeep
		}
	}

	prerequisites = {
		"tech_centralized_command"
	}

	show_tech_unlock_if = {
		NOT = { has_valid_civic = civic_reanimated_armies }
	}

	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = yes }
		}
		text = job_warrior_drone_effect_desc
	}
	triggered_desc = {
		trigger = {
			exists = owner
			owner = { is_gestalt = no }
		}
		text = job_soldier_effect_desc
	}
}

building_planetary_shield_generator = {
	base_buildtime = @btr_mili_b3_time
	base_cap_amount = 1
	position_priority = 100
	can_be_ruined = no
	category = army
	
	potential = {
		NOT = { has_planet_flag = planetary_station }
	}

	allow = {
		has_upgraded_capital = yes
		hidden_trigger = { owner = { is_ai = no } }
		if = {
			limit = {
				exists = orbital_defence
			}
			orbital_defence = {
				exists = starbase
				starbase = {
					NOR = {
						has_starbase_building = ring_orbital_shield_generator
						is_starbase_building_building = ring_orbital_shield_generator
					}
				}
			}
		}
	}

	destroy_trigger = {
		exists = owner
		owner = { is_ai = yes }
	}

	planet_modifier = {
		planet_orbital_bombardment_damage = -0.5
	}

	resources = {
		category = planet_buildings
		cost = {
			minerals = @btr_mili_b3_mineral_cost
			light_metals = @btr_mili_b3_light_metals_cost
			heavy_metals = @btr_mili_b3_heavy_metals_cost
			uranium = @btr_mili_b3_uranium_cost
		}
		upkeep = {
			energy = @btr_mili_b3_energy_upkeep
			uranium = @btr_mili_b3_uranium_upkeep
		}
	}

	prerequisites = {
		"tech_planetary_shield_generator"
	}
}
