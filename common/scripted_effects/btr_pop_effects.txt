generate_csec_pinnacle_pop = {
	while = {
		count = 1
		generate_asariPop = yes
	}
	every_owned_pop = { set_pop_flag = init_spawn }
	while = {
		count = 1
		generate_salarianPop = yes
	}
	every_owned_pop = { set_pop_flag = init_spawn }
	while = {
		count = 1
		generate_turianPop = yes
	}
}
