
create_asari_republics_construction_ship = {
	create_fleet = {
		effect = {
			set_owner = event_target:asari_republics
			create_ship = {
				name = random
				design = "NAME_Constructor"
			}
			set_location = {
				target = prev
				distance = 45
				angle = random
			}
			set_fleet_stance = evasive
		}
	}
}

create_asari_republics_science_ship = {
	event_target:asari_republics = {
		create_fleet = {
			set_owner = event_target:salarian_union
			effect = {
				create_ship = {
					name = random
					random_existing_design = science
				}
				set_location = {
					target = prev
					distance = 45
					angle = random
				}
				set_fleet_stance = evasive
			}
		}		
	}
}
