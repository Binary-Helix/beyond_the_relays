generate_asari_republics_country = {
	create_country = {
		type = default
		ignore_initial_colony_error = yes
		name = "Asari Republics"
		adjective = "PRESCRIPTED_adjective_asari_republics"
		species = event_target:asariSpecies
		name_list = btr_asari
		authority = auth_democratic
		ship_prefix = "ASV"
		origin = asari_rep_origin
		ethos = { 
			ethic = "ethic_fanatic_xenophile" 
			ethic = "ethic_egalitarian" 
		}
		civics = { 
			civic = asari_rep_civic_thes_dip_corp 
			civic = asari_rep_civic_democracy 
		}
		flag = {
			icon={
				category="beyondtherelays"
				file="asari_republics.dds"
			}
			background={
				category="backgrounds"
				file="flag_BG_42.dds"
			}
			colors={
				"blue"
				"blue"
				"null"
				"null"
			}
		}
		effect = {
			set_graphical_culture = mec_asari_01
			set_city_graphical_culture = mec_asari_01
			set_country_flag = asari_republics_ai
			set_country_flag = asari_republics
			set_country_flag = init_spawned
			set_country_flag = canon_map
			set_country_flag = btr_empire
			set_ai_personality = asari_republic
			save_global_event_target_as = asari_republics
			add_resource = {
				energy = 3500
				minerals = 2000
				alloys = 1500
				influence = 100
				food = 3500
				consumer_goods = 1500
				unity = 450
				volatile_motes = 200
				exotic_gases = 200
				rare_crystals = 700
			}
			add_modifier = {
				modifier = asari_rep_empire_modifier
				days = -1
			}
		}
	}
	random_country = {
		limit = { 
			OR = {
				has_country_flag = illium_corporate_interests_player
				has_country_flag = omega_empire_player
				has_country_flag = illium_corporate_interests_ai
				has_country_flag = omega_empire_ai
				has_country_flag = asari_republics_ai
				has_country_flag = asari_republics_player
			}
		}
		every_owned_pop_species = {
			limit = {
				is_species_class = ASA
			}
			set_species_homeworld = prevprev
		}
	}
	log = "Asari Republics AI Empire Created"
}

			skill = 1
			traits = {
				trait = random_trait
			}
		}				
		if = {
			limit = {
				NOT = {
					has_modifier = pirate_power
				}
			}
			add_modifier = {
				modifier = pirate_power
				days = -1
			}
		}
	}
	every_country = {
		establish_communications_no_message = event_target:pirate_country
	}
	create_fleet = {
		name = "Pirate Station"
		effect = {
			set_owner = event_target:pirate_country
			create_ship = {
				name = random

				graphical_culture = "pirate_01"
			}
			set_location = {
				target = prev
				distance = 6
				angle = 120
			}
		}
	}			
	create_fleet = {
		name = "Red's Raiders"
		effect = {
			set_owner = event_target:pirate_country
			set_fleet_flag = pirate_home_system_fleet

				count = 1
