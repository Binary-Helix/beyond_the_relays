					}
					set_timed_pop_flag = {
						flag = pop_convicted
						years = 20
					}
					resettle_pop = {
						pop = this
						planet = event_target:prison_station
					}
				}
			}
			
			create_message = {
				type = FELONY_TRANSPORT
				localization = FELONY_TRANSPORT_DESC
				days = 30
				target = THIS
				variable = {
					type = name
					localization = PLANET
					scope = this
				}

					type = name
					localization = PRISON

		weight = 20
	}
