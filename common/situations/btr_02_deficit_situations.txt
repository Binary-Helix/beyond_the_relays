		hidden_effect = {

	}

		custom_tooltip = deficit_situation_finish_tooltip

	monthly_progress = {
		base = 0
		modifier = {
			add = value:deficit_situation_progress_value|RESOURCE|lithium|

			owner = { has_actual_deficit = { RESOURCE = lithium } }

			subtract = 5
			desc = string_not_in_deficit_lithium

				resource_income_compare = {
					resource = lithium

			subtract = 1
			desc = string_resource_reserves_lithium

				has_negative_income_with_stockpile = { RESOURCE = lithium }
			}
		}

			owner = {
				OR = {
					is_hive_empire = yes
					is_catalytic_empire = no
					country_uses_consumer_goods = yes
				}
			}
		}

		ai_weight = {
			base = 0
			modifier = {
				add = 2
				current_stage = deficit_third
			}
			modifier = {
				add = 2
