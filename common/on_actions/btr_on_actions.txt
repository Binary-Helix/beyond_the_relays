on_game_start = {
	events = {
		btr_map_start.1 # Set Global Flag 'has_beyond_the_relays'
		btr_game_start.1 # Spawn special/hidden countries
		btr_game_start.2 # removes random planet modifiers from homeworlds and spawns #terraforming candidates
		btr_game_start.3 # Hidden Game Start Event that resets tutorial for advisor trigger purposes
		btr_map_start.102 # Create Real Space Nebula
		btr_map_start.103 # Create New Star Space
		btr_map_start.105 # Planet Rings!
		btr_map_start.107 # Remove Vanilla Nebula
		btr_map_start.109 # Generate Starting Subclasses
		btr_map_start.110 # Add High Gravity Modifier
		btr_map_start.111 # Add Dextro Modifier 
		btr_game_start.10 # Starbases and Borders
	}
}

# Triggers country_event for the defender upon victory
# Root = country, planet owner
# From = country, attack leader
# FromFrom = planet

	events = {
		planetarystations_colony_event.14
		planetarystations_colony_event.18
		planetarystations_colony_event.38
	}
}

# From = Bombarder

	}
}
