# Common utility events utilized by the empires 
# on_actions: 001 - 100

namespace = btr_common

country_event = {
	id = btr_common.001
	is_triggered_only = yes
	hide_window = yes

	trigger = {
		always = yes
	}

	immediate = {
		if = {
			limit = { from = { species = { has_trait = trait_asari_core } } }
			from = {
				add_trait = asari_leader_trait_long_view
				add_skill = 2
			}
			if = {
				limit = { from = { leader_class = general }}
				from = {
					add_trait = asari_leader_trait_commando
				}
			}
		}
		else_if = {
			limit = { 
				from = { 
					species = { has_trait = trait_turian_core }
					OR = {
						leader_class = general
						leader_class = admiral
					}
				} 
			}
			from = {
				add_trait = turian_leader_trait_extensive_training
			}
		}
		else_if = {
			limit = { 
				from = { 
					species = { has_trait = geth_trait_core }
				} 
			}
			from = {
				add_trait = geth_leader_trait_internal_consensus
			}
		}
		else_if = {
			limit = { 
				from = {
					species = { has_trait = human_trait_core }
				}
			}
			from = {
				add_trait = alliance_leader_trait_twice_bright
			} 
		}
		else_if = {
			limit = { 
				from = {
					species = { has_trait = yahg_trait_core }
				}
			}
			from = {
				add_trait = yahg_leader_trait_hyper_perceptiveness
			} 
		}

	}
}
