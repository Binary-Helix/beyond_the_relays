# Compatibility flag. Used to tell other mods that btr was installed.
namespace = btr_mod_flag
event = {
	id = btr_mod_flag.1
	hide_window = yes
	is_triggered_only = yes
	fire_only_once = no
	immediate = {
		set_global_flag = "btr_enabled_flag"
		btr_calculate_real_truce_years = yes
		if = {
			limit = {
				btr_is_startech_loaded = yes
			}
			set_global_flag = "btr_tech_fork"
		}
		if = {
			limit = {
				btr_is_startech_loaded = no
				has_global_flag = "btr_tech_fork"
			}
			remove_global_flag = "btr_tech_fork"
		}
		if = {
			limit = {
				btr_is_friendship_patch_loaded = no
				has_global_flag = "btr_friendship_flag"
			}
			remove_global_flag = "btr_friendship_flag"
		}
		if = {
			limit = {
				btr_is_mixed_fleet_loaded = no
				has_global_flag = "btr_mixed_fleet_flag"
			}
			remove_global_flag = "btr_mixed_fleet_flag"
		}
		if = {
			limit = {
				btr_is_nsc2_patch_loaded = no
				has_global_flag = "btr_nsc2_patch_flag"
			}
			remove_global_flag = "btr_nsc2_patch_flag"
		}
	}
}
