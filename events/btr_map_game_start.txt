#Create Real Space Nebula
event = {
	id = btr_map_start.102
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_global_flag = only_vanilla_nebula_setting }
	}
	
	immediate = {
		every_system = {
			limit = {
				has_star_flag = widow_system
			}
			create_bright_purple_nebula_1 = yes
			create_bright_purple_nebula_2 = yes
			create_faint_purple_nebula_1 = yes
			create_faint_purple_nebula_2 = yes
			create_faint_purple_nebula_3 = yes
			create_faint_purple_nebula_4 = yes
		}

			limit = {
				has_star_flag = tartarus_debris_field
			}
			create_thick_orange_nebula_1 = yes
			create_thick_orange_nebula_2 = yes
			create_thick_red_nebula_1 = yes
		}

		log = "btr_map_start.102 Real Space Nebulae Created"
	}

}

#Rings 
event = {
	id = btr_map_start.105
	title = OK
	desc = OK
	hide_window = yes
	
	is_triggered_only = yes
	
	immediate = {
		# Sol System 
		every_galaxy_planet = {
			limit = { has_planet_flag = jupiter_planet }
			save_event_target_as = planet_location_rings
			create_ambient_object = {
				type = jupiter_rings
				location = this
			}
			last_created_ambient_object = {
				set_location = {
					target = event_target:planet_location_rings
				}
			}
			set_planet_flag = has_rings_02_low
		}
		every_galaxy_planet = {
			limit = { has_planet_flag = saturn_planet }
			save_event_target_as = planet_location_rings
			create_ambient_object = {
				type = saturn_rings_01_large
				location = this
			}
			last_created_ambient_object = {
				set_location = {
					target = event_target:planet_location_rings
				}
			}
			set_planet_flag = has_rings_01_large
		}
		every_galaxy_planet = {
			limit = { has_planet_flag = uranus_planet }
			save_event_target_as = planet_location_rings
			create_ambient_object = {
				type = uranus_rings
				location = this
			}
			last_created_ambient_object = {
				set_location = {
					target = event_target:planet_location_rings
				}
			}
			set_planet_flag = has_rings_02_low
		}
		every_galaxy_planet = {
			limit = { has_planet_flag = neptune_planet }
			save_event_target_as = planet_location_rings
			create_ambient_object = {
				type = neptune_rings
				location = this
			}
			last_created_ambient_object = {
				set_location = {
					target = event_target:planet_location_rings
				}
			}
			set_planet_flag = has_rings_02_low
		}
		# Gas Giant
		every_galaxy_planet = {
			limit = {
				AND = {
					is_planet_class = "pc_gas_giant"
					has_ring = yes
					solar_system = { NOT = { has_star_flag = sol } }
					planet_size >= 25
				}
			}
			if = {
				limit = {
					is_hot_zone = no
				}
				set_gas_giant_big_rings = yes
			}
			else = {
				set_ring = no
			}
		}
		every_galaxy_planet = {
			limit = {
				AND = {
					is_planet_class = "pc_gas_giant"
					has_ring = yes
					solar_system = { NOT = { has_star_flag = sol } }
					planet_size < 25
				}
			}
			if = {
				limit = {
					is_hot_zone = no
				}
				set_gas_giant_small_rings = yes
			}
			else = {
				set_ring = no
			}
		}
		# Small Planet
		every_galaxy_planet = {
			limit = {
				AND = {
					NOT = { is_planet_class = "pc_gas_giant" }
					has_ring = yes
					planet_size >= 20
				}
			}
			if = {
				limit = {
					is_hot_zone = no
				}
				set_small_planet_big_rings = yes
			}
			else = {
				set_ring = no
			}
		}
		every_galaxy_planet = {
			limit = {
				AND = {
					NOT = { is_planet_class = "pc_gas_giant" }
					has_ring = yes
					planet_size < 20
				}
			}
			if = {
				limit = {
					is_hot_zone = no
				}
				set_small_planet_thin_rings = yes
			}
			else = {
				set_ring = no
			}
		}

		log = "btr_map_start.105 - Planet Rings Created"
	}
}

#Remove Vanilla Nebula
event = {
	id = btr_map_start.107
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		NOT = { has_global_flag = only_vanilla_nebula_setting }
	}
	
	immediate = {
		every_system = {
			limit = { is_inside_nebula = yes }
			random_system_ambient_object = { 
				limit = { is_vanilla_nebula = yes } 
				destroy_ambient_object = this
			}
		}

		log = "btr_map_start.107 - Vanilla Nebula removed"
	}
}

#Remove planet ring for megastructure build
event = {
	id = btr_map_start.108
	hide_window = yes
	
	is_triggered_only = yes
	
	trigger = {
		FROM = { is_megastructure_type = ring_world_2_intermediate }
	}
	
	immediate = {
		FROMFROM = {
			every_system_ambient_object = { 
				limit = { is_rs_rings_ambient_object = yes } 
				destroy_ambient_object = this
			}
		}

		log = "btr_map_start.108 - Removed Planet Ring for Megastructure"
	}
}
