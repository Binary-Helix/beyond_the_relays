event = {
	id = btr_game_start.4
	hide_window = yes
	is_triggered_only = yes
	immediate = {
		every_country = {
			if = {
				limit = {
					is_country_type = default
					is_country_type = neutral_faction
					is_country_type = rebel
					is_country_type = enclave
					is_country_type = tiyanki
					is_country_type = tiyanki_garrison
					is_country_type = amoeba
					is_country_type = amoeba_borderless
					is_country_type = amoeba_garrison
					is_country_type = crystal
					is_country_type = drone
					NOT = { has_country_flag = btr_empire }
					is_ai = yes
				}
				destroy_country = yes
				every_owned_pop = {	
					limit = { 
						NOT = { 
							 has_pop_flag = init_spawn 
						}
					}
					kill_pop = yes
				}
			}
			if = {
				limit = {
					has_country_flag = btr_empire
				}
				every_owned_pop = {
					limit = { NOT = { has_pop_flag = init_spawn } }
					kill_pop = yes
				}
				every_country = {
					every_owned_planet = {
						check_planet_employment = yes				
					}
				}
			}
			if = {
				limit = {
					OR = {
						has_country_flag = geth_consensus
						has_country_flag = disciples_of_nazara
					}
				}
				every_owned_pop = {
					change_species = event_target:gethSpecies
				}
			}

				limit = {
					has_country_flag = omega_empire

country_event = { #Courtesy of MEC Asari Civilizations
    id = btr_game_start.6
    hide_window = yes
    #fire_only_once = yes
	is_triggered_only = yes
	
    immediate = {
		# If any "canon" asari planets exist, save their owners
		random_country = {
			limit = {
				has_origin = asari_rep_origin
			}
			save_event_target_as = asari_thessia_country
			species = { save_event_target_as = asari_thessia_species }
		}
		random_country = {
			limit = {
				has_origin = omega_origin
			}
			save_event_target_as = asari_omega_country
			species = { save_event_target_as = asari_omega_species }
		}
		random_country = {
			limit = {
				has_origin = illium_origin
			}
			save_event_target_as = asari_illium_country
			species = { save_event_target_as = asari_illium_species }
		}
		# Check for the first "canon" asari species generated by the game.
		# This is to make sure it's folded into the main species on the list instead of being shown in the drop-down.
		every_country = {
			limit = {
				OR = {
					is_same_value = event_target:asari_thessia_country
					is_same_value = event_target:asari_omega_country
					is_same_value = event_target:asari_illium_country
				}
			}
			if = {
				limit = {
					NOT = { exists = event_target:asari_main_country }
				}
				save_global_event_target_as = asari_main_country
				species = { save_global_event_target_as = asari_main_species }
				# If Thessia exists, set it as the main asari species homeworld
				if = {
					limit = {
						exists = event_target:parnitha_system_thessia
					}
					event_target:asari_main_species = {
						set_species_homeworld = event_target:parnitha_system_thessia
					}
				}
			}
		}
		# Check for duplicate asari offshots and merge them all into one species
		every_country = {
			if = {
				limit = {
					exists = event_target:asari_main_country
					OR = {
						species = { is_same_value = event_target:asari_thessia_species }
						species = { is_same_value = event_target:asari_omega_species }
						species = { is_same_value = event_target:asari_illium_species }
					}
				}
				change_dominant_species = {
					species = event_target:asari_main_species
					change_all = yes
				}
			}
		}
    }
}
