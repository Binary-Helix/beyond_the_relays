		portrait = root.owner_main_species

			limit = {

			}
			btr_menu_truce_2 = {
				set_global_flag = btr_global_disable_will_not_declare_war_until
				remove_global_flag = btr_menu_truce_2
				set_global_flag = btr_menu_truce_3
			}
			btr_menu_truce_3 = {
				remove_global_flag = btr_player_will_not_declare_war_until
				remove_global_flag = btr_global_disable_will_not_declare_war_until
				remove_global_flag = btr_menu_truce_3
				set_global_flag = btr_menu_truce_1
			}
			default = {
				remove_global_flag = btr_player_will_not_declare_war_until
				remove_global_flag = btr_global_disable_will_not_declare_war_until
				set_global_flag = btr_menu_truce_1
				remove_global_flag = btr_menu_truce_2
				remove_global_flag = btr_menu_truce_3
			}
		}
		hidden_effect = {
			country_event = {
				id = btr_menu.1
			}
		}
	}
	option = {
		name = btr_menu.1.truce.settings
		custom_tooltip = btr_menu.1.truce.settings.tooltip

			country_event = {
				id = btr_menu.2
